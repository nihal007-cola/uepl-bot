function doPost(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("UEPL_DB"); if (!sheet) throw new Error("Sheet not found"); const data = JSON.parse(e.postData.contents); // 🔥 HANDLE BASE64 IMAGE let base64 = data.image.split(",")[1]; let blob = Utilities.newBlob( Utilities.base64Decode(base64), "image/jpeg", data.item + ".jpg" ); // 🔥 SAVE TO DRIVE let file = DriveApp.createFile(blob); file.setSharing(DriveApp.Access.ANYONE, DriveApp.Permission.VIEW); let driveUrl = "https://drive.google.com/uc?id=" + file.getId(); // 🔥 WRITE TO SHEET (MATCH HTML FIELDS) sheet.appendRow([ data.item, data.count, data.construction, data.composition, data.weight, data.width, data.availability, driveUrl, new Date() ]); return ContentService .createTextOutput(JSON.stringify({ status: "ok" })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("UEPL_DB"); if (!sheet) return ContentService.createTextOutput("SHEET NOT FOUND"); const data = sheet.getDataRange().getValues(); let result = []; for (let i = 1; i < data.length; i++) { result.push({ item: data[i][0], count: data[i][1], construction: data[i][2], composition: data[i][3], weight: data[i][4], width: data[i][5], availability: data[i][6], image: data[i][7] }); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
